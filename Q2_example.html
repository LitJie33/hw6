資料視覺化

# Write your code here
pl <- ggplot(df) +
  geom_histogram(aes(x = carat), 
                 alpha = 0.6, binwidth = 0.1) +
  geom_freqpoly(aes(carat, color = cut), 
                binwidth = 0.1)

# 請務必印出圖片
pl
